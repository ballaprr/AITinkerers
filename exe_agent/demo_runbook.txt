        ACTION: cli command="docker logs $(docker ps -q --filter 'name=postgres' --latest 2>/dev/null) --tail 20 2>/dev/null || echo 'No postgres container found'"
        ACTION: screenshot name_prefix="db_logs"

    Document Results:
        ACTION: cli command="echo '=== Database Investigation Results ===' > investigation_results.txt && date >> investigation_results.txt && echo '\n=== System Status ===' >> investigation_results.txt && echo '\n=== Database Status ===' >> investigation_results.txt && docker ps | grep postgres >> investigation_results.txt || echo 'No postgres container found' >> investigation_results.txt && echo '\n=== Investigation Complete ===' >> investigation_results.txt"
