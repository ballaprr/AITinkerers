Database Issue Investigation:
    Check System Resources:
        Check Memory Usage
        ACTION: cli command="timeout 5 top -l 1 -n 5 | head -n 10"
        ACTION: screenshot name_prefix="memory_status"
        
        Check Disk Space
        ACTION: cli command="timeout 5 df -h | head -n 10"
        ACTION: screenshot name_prefix="disk_status"

    Check Database Status:
        Check Docker Container Status
        ACTION: cli command="timeout 5 docker ps | grep postgres || true"
        ACTION: screenshot name_prefix="db_container_status"
        
        Check Database Logs
        ACTION: cli command="timeout 5 docker logs $(docker ps -q --filter 'name=postgres' --latest 2>/dev/null) --tail 20 2>/dev/null || echo 'No postgres container found'"
        ACTION: screenshot name_prefix="db_logs"

    Check Database Metrics:
        Get Connection Count
        ACTION: cli command="timeout 5 docker exec $(docker ps -q --filter 'name=postgres' --latest 2>/dev/null) psql -U postgres -c 'SELECT count(*) FROM pg_stat_activity;' 2>/dev/null || echo 'No postgres container found'"
        
        Check Transaction Status
        ACTION: cli command="timeout 5 docker exec $(docker ps -q --filter 'name=postgres' --latest 2>/dev/null) psql -U postgres -c 'SELECT state, count(*) FROM pg_stat_activity GROUP BY state;' 2>/dev/null || echo 'No postgres container found'"

    Search Known Issues:
        Check PostgreSQL Documentation
        ACTION: go_to url=https://www.postgresql.org/docs/current/
        ACTION: search query="connection pooling best practices" search_box_selector=".searchbox"
        ACTION: screenshot name_prefix="postgres_docs"

    Resolve Issues:
        Restart Database Container
        ACTION: cli command="timeout 10 docker restart $(docker ps -q --filter 'name=postgres' --latest 2>/dev/null) || echo 'No postgres container found'"
        
        Verify Database Recovery
        ACTION: cli command="timeout 5 docker exec $(docker ps -q --filter 'name=postgres' --latest 2>/dev/null) pg_isready 2>/dev/null || echo 'No postgres container found'"
        ACTION: screenshot name_prefix="db_recovery"

    Document Results:
        Export Diagnostic Information
        ACTION: cli command="echo '=== Database Investigation Results ===' > investigation_results.txt"
        ACTION: cli command="date >> investigation_results.txt"
        ACTION: cli command="echo '\n=== System Status ===' >> investigation_results.txt"
        ACTION: cli command="timeout 5 top -l 1 -n 5 | head -n 10 >> investigation_results.txt"
        ACTION: cli command="echo '\n=== Database Status ===' >> investigation_results.txt"
        ACTION: cli command="timeout 5 docker ps | grep postgres >> investigation_results.txt || echo 'No postgres container found' >> investigation_results.txt"
        ACTION: cli command="echo '\n=== Investigation Complete ===' >> investigation_results.txt"
        
        Save Screenshots
        ACTION: cli command="mkdir -p diagnostic_results && mv *.png diagnostic_results/ 2>/dev/null || echo 'No screenshots to move'"
        
END_RUNBOOK
